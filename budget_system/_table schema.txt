-- BSA Budget System
-- SQL Server schema
-- Target: Microsoft SQL Server 2019+
--
-- Notes:
-- 1. PostgreSQL array fields are stored as JSON text in SQL Server.
-- 2. JSON columns use NVARCHAR(MAX) plus ISJSON checks.
-- 3. This file is aligned to the current SQLAlchemy models and includes bsa_actual.

/* -----------------------------------------------------------
   Table: dbo.bsa_main
----------------------------------------------------------- */
CREATE TABLE dbo.bsa_main (
    id BIGINT IDENTITY(1,1) NOT NULL,
    version NVARCHAR(50) NULL,
    data_type NVARCHAR(50) NULL,
    under_ops_control NVARCHAR(10) NULL,
    ccgl NVARCHAR(50) NULL,
    glc NVARCHAR(50) NULL,
    cc NVARCHAR(50) NULL,
    non_controllable NVARCHAR(50) NULL,
    area NVARCHAR(50) NULL,
    dept NVARCHAR(50) NULL,
    dept_group NVARCHAR(50) NULL,
    dept_ppt NVARCHAR(300) NULL,
    category NVARCHAR(100) NULL,
    discretionary NVARCHAR(50) NULL,
    at_var DECIMAL(18,4) NULL,
    self_study_var DECIMAL(18,4) NULL,
    spends_control NVARCHAR(10) NULL,
    iecs_view NVARCHAR(10) NULL,
    levels NVARCHAR(300) NULL,
    accounts NVARCHAR(300) NULL,
    budgeter NVARCHAR(100) NULL,
    baseline_adjustment DECIMAL(18,2) NULL,
    created_at DATETIME2(0) NOT NULL
        CONSTRAINT DF_bsa_main_created_at DEFAULT SYSDATETIME(),
    updated_at DATETIME2(0) NOT NULL
        CONSTRAINT DF_bsa_main_updated_at DEFAULT SYSDATETIME(),
    CONSTRAINT PK_bsa_main PRIMARY KEY CLUSTERED (id)
);
GO

CREATE INDEX IX_bsa_main_version ON dbo.bsa_main (version);
GO
CREATE INDEX IX_bsa_main_area ON dbo.bsa_main (area);
GO
CREATE INDEX IX_bsa_main_dept ON dbo.bsa_main (dept);
GO
CREATE INDEX IX_bsa_main_dept_group ON dbo.bsa_main (dept_group);
GO
CREATE INDEX IX_bsa_main_category ON dbo.bsa_main (category);
GO
CREATE INDEX IX_bsa_main_budgeter ON dbo.bsa_main (budgeter);
GO
CREATE INDEX IX_bsa_main_cc ON dbo.bsa_main (cc);
GO
CREATE INDEX IX_bsa_main_version_dept ON dbo.bsa_main (version, dept);
GO
CREATE INDEX IX_bsa_main_version_budgeter ON dbo.bsa_main (version, budgeter);
GO

/* -----------------------------------------------------------
   Table: dbo.bsa_permission
----------------------------------------------------------- */
CREATE TABLE dbo.bsa_permission (
    id INT IDENTITY(1,1) NOT NULL,
    user_mail NVARCHAR(320) NOT NULL,
    user_role NVARCHAR(50) NOT NULL,
    user_area NVARCHAR(MAX) NULL
        CONSTRAINT DF_bsa_permission_user_area DEFAULT N'[]',
    CONSTRAINT PK_bsa_permission PRIMARY KEY CLUSTERED (id),
    CONSTRAINT UQ_bsa_permission_user_mail_role UNIQUE (user_mail, user_role),
    CONSTRAINT CK_bsa_permission_user_area_json
        CHECK (user_area IS NULL OR ISJSON(user_area) = 1)
);
GO

/* -----------------------------------------------------------
   Table: dbo.bsa_submission_status
----------------------------------------------------------- */
CREATE TABLE dbo.bsa_submission_status (
    id BIGINT IDENTITY(1,1) NOT NULL,
    version NVARCHAR(50) NOT NULL,
    dept_ppt NVARCHAR(300) NOT NULL,
    status NVARCHAR(20) NOT NULL
        CONSTRAINT DF_bsa_submission_status_status DEFAULT N'not_started',
    submitted_at DATETIME2(0) NULL,
    updated_at DATETIME2(0) NOT NULL
        CONSTRAINT DF_bsa_submission_status_updated_at DEFAULT SYSUTCDATETIME(),
    submitted_by NVARCHAR(MAX) NULL
        CONSTRAINT DF_bsa_submission_status_submitted_by DEFAULT N'[]',
    CONSTRAINT PK_bsa_submission_status PRIMARY KEY CLUSTERED (id),
    CONSTRAINT UQ_bsa_submission_status_version_dept_ppt UNIQUE (version, dept_ppt),
    CONSTRAINT CK_bsa_submission_status_status
        CHECK (status IN (N'not_started', N'editing', N'under_review', N'complete')),
    CONSTRAINT CK_bsa_submission_status_submitted_by_json
        CHECK (submitted_by IS NULL OR ISJSON(submitted_by) = 1)
);
GO

CREATE INDEX IX_bsa_submission_status_version ON dbo.bsa_submission_status (version);
GO
CREATE INDEX IX_bsa_submission_status_dept_ppt ON dbo.bsa_submission_status (dept_ppt);
GO
